\end{block} \end{frame} \hypertarget{optimization-by-software-2.0}{% \section{Optimization by ``Software 2.0''}\label{optimization-by-software-2.0}} \begin{frame}{Intro} \protect\hypertarget{intro}{} \begin{itemize} \item Major downsides like difficult optimization of code and human error in classical software development \item Doesn't base on declarative programming and tries to learn the desired functionality with a approximate base net \item Enabled by development of Neural Networks in last 20 years allowing \textgreater{}100 layers deep networks \item Program space is restricted for future training (backpropagation, gradient descent) \item Many real world problems easier to detect desirable behaviour than to write a specific program \end{itemize} \end{frame} \begin{frame}{Advantages} \protect\hypertarget{advantages-1}{} \begin{block}{Higher Portability} \begin{itemize} \tightlist \item Smaller operation set \item Matrix Multiplication and thresholding at zero required \item Small instruction set of chips with pretrained nets allows for cheap and specialised hardware \end{itemize} \end{block} \begin{block}{Better Performance} \begin{itemize} \tightlist \item Allows for better performance and correctness predictions because closer implementation in hardware less core primitives are needed \item Modules can be introduced to a single module reducing communication overhead by sacrificing clarity of separation, which is due to the human unlike nature of S2.0 sacrificed beforehand either way \item Well trained neural nets outperform code implementation \end{itemize} \end{block} \end{frame} \begin{frame}{Advantages} \protect\hypertarget{advantages-2}{} \begin{figure} \centering \includegraphics[width=\textwidth,height=0.4\textheight]{so2.png} \caption{\emph{https://medium.com/@karpathy/software-2-0-a64152b37c35, Karpathy}} \end{figure} \begin{block}{Better Runtime Predictability} \begin{itemize} \tightlist \item Requires same amount of memory each iteration → low probability of infinite loops or locks \item Speed well adjustable → speed can easily improved by reducing performance \end{itemize} master thesis taranpreet kaur table note \end{block} \end{frame} \begin{frame}{Disadvantages} \protect\hypertarget{disadvantages-1}{} \begin{block}{Unintuitivity} \begin{itemize} \tightlist \item Can be treated as different new paradigm → requires rethinking of development style \item Even though the network may work well, for humans difficult to understand \item Developing S2.0 is unintuitive and not well developed \item Requires manually curating, maintaining, cleaning and labeling of datasets \item May not be applicable easily to all problems \end{itemize} \end{block} \end{frame} \begin{frame}{Disadvantages} \protect\hypertarget{disadvantages-2}{} \begin{block}{Nonrecognizable errors} \begin{itemize} \tightlist \item Errors may occur unpredictable \item Can silently fail due to changed biases (hard to track since a large amount of them are being trained) \end{itemize} \end{block} \begin{block}{Lack of Tools} \begin{itemize} \tightlist \item No tools currently exist that support the development process like IDE, highlighting and alike as for classical software \end{itemize} \end{block} \begin{block}{Low Experience} \begin{itemize} \tightlist \item Effective in some implementations but not as general approach \end{itemize} \end{block} \end{frame} \begin{frame}{Conclusion} \protect\hypertarget{conclusion}{} \begin{block}{Optimization} \begin{itemize} \tightlist \item Allows for greater optimization of specific complex problems \item e.g.~Cuttlefish achieves 7.5x speedup to other query optimizers \emph{Cuttlefish: A Lightweight Primitive for Adaptive Query Processing} \end{itemize} \end{block} \begin{block}{Development difficult} \begin{itemize} \tightlist \item No Tools and unexperienced developers \item Generally low usage experience \end{itemize} \end{block} \end{frame}
